\begin{equation}
a_{i}
\equiv \frac{1 }{2 } \left(c_{i,A} + \mathrm{i} c_{i,B} \right),\quad
a^\dag_{i}
\equiv \frac{1 }{2 } \left(c_{i,A} - \mathrm{i} c_{i,B} \right),\quad
\end{equation}
$$

$$
\begin{equation}
c_{i,A}
=a_{i} + a^\dag_{i},\quad
c_{\bm{r},B}
=\frac{1 }{\mathrm{i} } \left(a_{i} - a^\dag_{i} \right),
\end{equation}
$$

\begin{equation}
\begin{aligned}
\alpha_i

\bm{r}\in A
\end{equation}

\begin{equation}
\chi^\dag_{\bm{r},\alpha}

\bm{r}\in A
\end{equation}
