### 实空间TB哈密顿量与k空间哈密顿量

$$
\begin{equation}
H_{\mathrm{TB}}
=H_{\mathrm{hop}} + H_{\mathrm{on-site}}
\end{equation}
$$

#### 实空间hopping项

$$
\begin{equation}
\begin{aligned}
H_{\mathrm{hop}}
&=\sum_{\bm{r},\mu} \sum_{\bm{r}',\nu} d^\dag_{\bm{r},\mu} \Braket{0_\chi,0_{\alpha(\bm{r},\mu)} | d_{\bm{r},\mu} H_h d^\dag_{\bm{r}',\nu} | 0_\chi,0_{\alpha(\bm{r}',\nu)} } d_{\bm{r}',\nu} \\
(\text{最近邻})&=\sum_{\bm{r}} \bigg[ d^\dag_{\bm{r},x} \Braket{0_\chi,0_{\alpha(\bm{r},x)} | d_{\bm{r},x} H_h d^\dag_{\bm{r},y} | 0_\chi,0_{\alpha(\bm{r},y)} } d_{\bm{r},y} \\
&+d^\dag_{\bm{r},z} \Braket{0_\chi,0_{\alpha(\bm{r},z)} | d_{\bm{r},z} H_h d^\dag_{\bm{r},x} | 0_\chi,0_{\alpha(\bm{r},x)} } d_{\bm{r},x} \\
&+d^\dag_{\bm{r},y} \Braket{0_\chi,0_{\alpha(\bm{r},y)} | d_{\bm{r},y} H_h d^\dag_{\bm{r},z} | 0_\chi,0_{\alpha(\bm{r},z)} } d_{\bm{r},z} \\
&+d^\dag_{\bm{r+a_2},x} \Braket{0_\chi,0_{\alpha(\bm{r+a_2},x)} | d_{\bm{r+a_2},x} H_h d^\dag_{\bm{r-a_1+a_2},y} | 0_\chi,0_{\alpha(\bm{r-a_1+a_2},y)} } d_{\bm{r-a_1+a_2},y} \\
&+d^\dag_{\bm{r},z} \Braket{0_\chi,0_{\alpha(\bm{r},z)} | d_{\bm{r},z} H_h d^\dag_{\bm{r+a_2},x} | 0_\chi,0_{\alpha(\bm{r+a_2},x)} } d_{\bm{r+a_2},x} \\
&+d^\dag_{\bm{r-a_1+a_2},y} \Braket{0_\chi,0_{\alpha(\bm{r-a_1+a_2},y)} | d_{\bm{r-a_1+a_2},y} H_h d^\dag_{\bm{r},z} | 0_\chi,0_{\alpha(\bm{r},z)} } d_{\bm{r},z} \bigg] \\
&+\mathrm{H.c.} \\
\end{aligned}
\end{equation}
$$

$$
\begin{equation}
\begin{aligned}
H_{\mathrm{hop}}
&=\sum_{\bm{r}} \bigg[ d^\dag_{\bm{r},x} T^A_{xy} d_{\bm{r},y} + d^\dag_{\bm{r},z} T^A_{zx} d_{\bm{r},x} + d^\dag_{\bm{r},y} T^A_{yz} d_{\bm{r},z} \\
&+d^\dag_{\bm{r+a_2},x} T^B_{xy} d_{\bm{r-a_1+a_2},y} + d^\dag_{\bm{r},z} T^B_{zx} d_{\bm{r+a_2},x} + d^\dag_{\bm{r-a_1+a_2},y} T^B_{yz} d_{\bm{r},z} \bigg] \\
&+\mathrm{H.c.} \\
&=\sum_{\bm{r}} \bigg[ d^\dag_{\bm{r},x} T^A_{xy} d_{\bm{r},y} + d^\dag_{\bm{r},y} T^A_{yz} d_{\bm{r},z} + d^\dag_{\bm{r},z} T^A_{zx} d_{\bm{r},x} \\
&+d^\dag_{\bm{r+a_2},x} T^B_{xy} d_{\bm{r-a_1+a_2},y} + d^\dag_{\bm{r-a_1+a_2},y} T^B_{yz} d_{\bm{r},z} + d^\dag_{\bm{r},z} T^B_{zx} d_{\bm{r+a_2},x} \bigg] \\
&+\mathrm{H.c.} \\
\end{aligned}
\end{equation}
$$

#### k空间hopping项

- $T^A_{xy} $

$$
\begin{equation}
\begin{aligned}
\sum_{\bm{r}} d^\dag_{\bm{r},x} T^A_{xy} d_{\bm{r},y} + \mathrm{H.c.}
&=\sum_{\bm{r}} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}} \mathrm{e}^{-\mathrm{i}\bm{k}\cdot \bm{r}} d^\dag_{\bm{k},x} \right) T^A_{xy} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}'} \mathrm{e}^{\mathrm{i}\bm{k}'\cdot \bm{r}} d_{\bm{k}',y} \right) + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} \sum_{\bm{r}} \mathrm{e}^{\mathrm{i}\left(\bm{k}'-\bm{k} \right)\cdot \bm{r}} d^\dag_{\bm{k},x} T^A_{xy} d_{\bm{k}',y} + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} N \delta_{\bm{k},\bm{k}'} d^\dag_{\bm{k},x} T^A_{xy} d_{\bm{k}',y} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} d^\dag_{\bm{k},x} T^A_{xy} d_{\bm{k},y} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \left[d^\dag_{\bm{k},x} T^A_{xy} d_{\bm{k},y} + d^\dag_{\bm{k},y} \left(T^A_{xy} \right)^\dag d_{\bm{k},x} \right] \\
\end{aligned}
\end{equation}
$$

- $T^A_{yz} $

$$
\begin{equation}
\begin{aligned}
\sum_{\bm{r}} d^\dag_{\bm{r},y} T^A_{yz} d_{\bm{r},z} + \mathrm{H.c.}
&=\sum_{\bm{k}} \left[d^\dag_{\bm{k},y} T^A_{yz} d_{\bm{k},z} + d^\dag_{\bm{k},z} \left(T^A_{yz} \right)^\dag d_{\bm{k},y} \right]
\end{aligned}
\end{equation}
$$

- $T^A_{zx} $

$$
\begin{equation}
\begin{aligned}
\sum_{\bm{r}} d^\dag_{\bm{r},z} T^A_{zx} d_{\bm{r},x} + \mathrm{H.c.}
&=\sum_{\bm{k}} \left[d^\dag_{\bm{k},z} T^A_{zx} d_{\bm{k},x} + d^\dag_{\bm{k},x} \left(T^A_{zx} \right)^\dag d_{\bm{k},z} \right]
\end{aligned}
\end{equation}
$$

- $T^B_{xy} $

$$
\begin{equation}
\begin{aligned}
\sum_{\bm{r}} d^\dag_{\bm{r+a_2},x} T^B_{xy} d_{\bm{r-a_1+a_2},y} + \mathrm{H.c.}
&=\sum_{\bm{r}} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}} \mathrm{e}^{-\mathrm{i}\bm{k}\cdot\left(\bm{r}+\bm{a}_2 \right)} d^\dag_{\bm{k},x} \right) T^B_{xy} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}'} \mathrm{e}^{\mathrm{i}\bm{k}'\cdot\left(\bm{r}-\bm{a}_1+\bm{a}_2 \right)} d_{\bm{k}',y} \right) + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} \sum_{\bm{r}} \mathrm{e}^{\mathrm{i}\left(\bm{k}'-\bm{k} \right)\cdot\bm{r}} \mathrm{e}^{\mathrm{i}\left(-\bm{k}\cdot\bm{a}_2 - \bm{k}'\cdot\bm{a}_1 + \bm{k}'\cdot\bm{a}_2 \right)} d^\dag_{\bm{k},x} T^B_{xy} d_{\bm{k}',y} + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} N \delta_{\bm{k},\bm{k}'} \mathrm{e}^{\mathrm{i}\left(-\bm{k}\cdot\bm{a}_2 - \bm{k}'\cdot\bm{a}_1 + \bm{k}'\cdot\bm{a}_2 \right)} d^\dag_{\bm{k},x} T^B_{xy} d_{\bm{k}',y} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \mathrm{e}^{\mathrm{i}\left(-\bm{k}\cdot\bm{a}_2 - \bm{k}\cdot\bm{a}_1 + \bm{k}\cdot\bm{a}_2 \right)} d^\dag_{\bm{k},x} T^B_{xy} d_{\bm{k},y} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \mathrm{e}^{-\mathrm{i} \bm{k}\cdot\bm{a}_1 } d^\dag_{\bm{k},x} T^B_{xy} d_{\bm{k},y} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \left[\mathrm{e}^{-\mathrm{i} \bm{k}\cdot\bm{a}_1 } d^\dag_{\bm{k},x} T^B_{xy} d_{\bm{k},y} + \mathrm{e}^{\mathrm{i} \bm{k}\cdot\bm{a}_1 } d^\dag_{\bm{k},y} \left(T^B_{xy} \right)^\dag d_{\bm{k},x} \right] \\
\end{aligned}
\end{equation}
$$

- $T^B_{yz} $

$$
\begin{equation}
\begin{aligned}
\sum_{\bm{r}} d^\dag_{\bm{r-a_1+a_2},y} T^B_{yz} d_{\bm{r},z} + \mathrm{H.c.}
&=\sum_{\bm{r}} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}} \mathrm{e}^{-\mathrm{i}\bm{k}\cdot\left(\bm{r}-\bm{a}_1+\bm{a}_2 \right)} d^\dag_{\bm{k},y} \right) T^B_{yz} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}'} \mathrm{e}^{\mathrm{i}\bm{k}'\cdot\bm{r} } d_{\bm{k}',z} \right) + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} \sum_{\bm{r}} \mathrm{e}^{\mathrm{i}\left(\bm{k}'-\bm{k} \right)\cdot\bm{r}} \mathrm{e}^{\mathrm{i}\bm{k}\cdot\left(\bm{a}_1 - \bm{a}_2 \right)} d^\dag_{\bm{k},y} T^B_{yz} d_{\bm{k}',z} + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} N \delta_{\bm{k},\bm{k}'} \mathrm{e}^{\mathrm{i}\bm{k}\cdot\left(\bm{a}_1 - \bm{a}_2 \right)} d^\dag_{\bm{k},y} T^B_{yz} d_{\bm{k}',z} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \mathrm{e}^{\mathrm{i}\bm{k}\cdot\left(\bm{a}_1 - \bm{a}_2 \right)} d^\dag_{\bm{k},y} T^B_{yz} d_{\bm{k},z} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \mathrm{e}^{\mathrm{i}\bm{k}\cdot\left(\bm{a}_1 - \bm{a}_2 \right)} d^\dag_{\bm{k},y} T^B_{yz} d_{\bm{k},z} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \left[\mathrm{e}^{\mathrm{i}\bm{k}\cdot\left(\bm{a}_1 - \bm{a}_2 \right)} d^\dag_{\bm{k},y} T^B_{yz} d_{\bm{k},z} + \mathrm{e}^{-\mathrm{i}\bm{k}\cdot\left(\bm{a}_1 - \bm{a}_2 \right)} d^\dag_{\bm{k},z} \left(T^B_{yz} \right)^\dag d_{\bm{k},y} \right] \\
\end{aligned}
\end{equation}
$$

- $T^B_{zx} $

$$
\begin{equation}
\begin{aligned}
\sum_{\bm{r}} d^\dag_{\bm{r},z} T^B_{zx} d_{\bm{r+a_2},x} + \mathrm{H.c.}
&=\sum_{\bm{r}} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}} \mathrm{e}^{-\mathrm{i}\bm{k}\cdot \bm{r}} d^\dag_{\bm{k},z} \right) T^B_{zx} \left(\frac{1 }{\sqrt{N} } \sum_{\bm{k}'} \mathrm{e}^{\mathrm{i}\bm{k}'\cdot \left(\bm{r}+\bm{a}_2 \right)} d_{\bm{k}',x} \right) + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} \sum_{\bm{r}} \mathrm{e}^{\mathrm{i}\left(\bm{k}'-\bm{k} \right)\cdot \bm{r}} \mathrm{e}^{\mathrm{i}\bm{k}'\cdot\bm{a}_2} d^\dag_{\bm{k},z} T^B_{zx} d_{\bm{k}',x} + \mathrm{H.c.} \\
&=\frac{1 }{N } \sum_{\bm{k},\bm{k}'} N \delta_{\bm{k},\bm{k}'} \mathrm{e}^{\mathrm{i}\bm{k}'\cdot\bm{a}_2} d^\dag_{\bm{k},z} T^B_{zx} d_{\bm{k}',x} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \mathrm{e}^{\mathrm{i}\bm{k}\cdot\bm{a}_2} d^\dag_{\bm{k},z} T^B_{zx} d_{\bm{k},x} + \mathrm{H.c.} \\
&=\sum_{\bm{k}} \left[\mathrm{e}^{\mathrm{i}\bm{k}\cdot\bm{a}_2} d^\dag_{\bm{k},z} T^B_{zx} d_{\bm{k},x} + \mathrm{e}^{-\mathrm{i}\bm{k}\cdot\bm{a}_2} d^\dag_{\bm{k},x} \left(T^B_{zx} \right)^\dag d_{\bm{k},z} \right] \\
\end{aligned}
\end{equation}
$$

$$
\begin{equation}
\begin{aligned}

#### 实空间on-site项

$$
\begin{equation}
H_c\left(0 \right)
=E_0\left(0 \right) + \sum_i \varepsilon_i(0) \alpha^\dag_i(0) \alpha_i(0)
\end{equation}
$$

$$
\begin{equation}
H_c\left(\bm{r},\mu \right)
=E_0\left(\mu \right) + \sum_i \varepsilon_i\left(\mu \right) \alpha_i^\dag\left(\bm{r},\mu \right) \alpha_i\left(\bm{r},\mu \right)
\end{equation}
$$

$$
\begin{equation}
E_i\left(\mu \right)
\equiv E_0\left(\mu \right) + \varepsilon_i\left(\mu \right) - E_0(0)
\end{equation}
$$

$$
\begin{equation}
E(\mu)
\equiv \mathrm{diag}(E_i(\mu))
\end{equation}
$$

$$
\begin{equation}
\begin{aligned}
H_{\mathrm{on-site}}
&=\sum_{\bm{r},\mu} \sum_{\bm{r}',\nu} d^\dag_{\bm{r},\mu} \Braket{0_\chi,0_{\alpha(\bm{r},\mu)} | d_{\bm{r},\mu} H_0 d^\dag_{\bm{r}',\nu} | 0_\chi,0_{\alpha(\bm{r}',\nu)} } d_{\bm{r}',\nu} \\
&=\sum_{\bm{r},\mu} d^\dag_{\bm{r},\mu} \Braket{0_\chi,0_{\alpha(\bm{r},\mu)} | d_{\bm{r},\mu} H_0 d^\dag_{\bm{r},\mu} | 0_\chi,0_{\alpha(\bm{r},\mu)} } d_{\bm{r},\mu} \\
&=\sum_{\bm{r},\mu} d^\dag_{\bm{r},\mu} \Braket{0_\chi,0_{\alpha(\bm{r},\mu)} | d_{\bm{r},\mu} H_c(\bm{r},\mu) d^\dag_{\bm{r},\mu} | 0_\chi,0_{\alpha(\bm{r},\mu)} } d_{\bm{r},\mu} \\
&=\sum_{\bm{r},\mu} d^\dag_{\bm{r},\mu} \Braket{0_\chi,0_{\alpha(\bm{r},\mu)} | d_{\bm{r},\mu} \left(E_0\left(\mu \right) + \sum_i \varepsilon_i\left(\mu \right) \alpha_i^\dag\left(\bm{r},\mu \right) \alpha_i\left(\bm{r},\mu \right) \right) d^\dag_{\bm{r},\mu} | 0_\chi,0_{\alpha(\bm{r},\mu)} } d_{\bm{r},\mu} \\
&=\sum_{\bm{r},\mu} d^\dag_{\bm{r},\mu}  \mathrm{diag}\left[E_0(\mu) + \varepsilon_1(\mu),E_0(\mu) + \varepsilon_2(\mu),\cdots \right] d_{\bm{r},\mu} \\
\text{所有能级同时减去}E_0(0)&=\sum_{\bm{r},\mu} d^\dag_{\bm{r},\mu}  \mathrm{diag}\left[E_0(\mu)-E_0(0) + \varepsilon_1(\mu),E_0(\mu)-E_0(0) + \varepsilon_2(\mu),\cdots \right] d_{\bm{r},\mu} \\
&\equiv \sum_{\bm{r},\mu} d^\dag_{\bm{r},\mu} E(\mu) d_{\bm{r},\mu}
\end{aligned}
\end{equation}
$$

#### 高对称点

$$
\begin{equation}
\bm{a}_1 = \left(a , 0 \right),\quad
\bm{a}_2 = \left(a/2 , \sqrt{3}a/2 \right)
\end{equation}
$$

$\bm{a}_1,\bm{a}_2 $ 二维平面单位法向量

$$
\begin{equation}
\bm{n}
=\frac{\bm{a}_1\times\bm{a}_2 }{\left|\bm{a}_1\times\bm{a}_2 \right| } 
\end{equation}
$$

倒格子基矢

$$
\begin{equation}
\bm{b}_1
=2\pi \frac{\bm{a}_2\times\bm{n} }{\bm{a}_1\cdot\left(\bm{a}_2\times\bm{n} \right) } ,\quad
\bm{b}_2
=2\pi \frac{\bm{n}\times\bm{a}_1 }{\bm{a}_2\cdot\left(\bm{n}\times\bm{a}_1 \right) }
\end{equation}
$$

$$
\begin{equation}
\bm{b}_1
=2\pi \frac{\bm{a}_2\times\bm{n} }{\left|\bm{a}_2\times\bm{n} \right| } \frac{1 }{a \cos\left(\pi/6 \right) } 
=\frac{4\pi }{\sqrt{3} a}  \frac{\bm{a}_2\times\bm{n} }{\left|\bm{a}_2\times\bm{n} \right| }
=\frac{4\pi }{\sqrt{3} a} \left(\sqrt{3}/2 , -1/2 \right)
=\left(2\pi/a , -2\pi/\sqrt{3}a \right)
\end{equation}
$$

$$
\begin{equation}
\bm{b}_2
